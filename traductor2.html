"""Crar un programa en python, que permita traducir un texto en español,
a 5 idiomas seleccionar "i" para traducir a ingles, "f" en frances, a aleman,
 "y" italiano, "p" portugues, y si preciona "x"que cierre el programa."""

from deep_translator import GoogleTranslator


while True:
    print("i. ingles")
    print("f. frances")
    print("y. italiano")
    print("a. aleman")
    print("p. portugues")
    print("x. para cerrar")
    seleccion = input("Selecciona un idioma de las que esta en la lista.")





    texto = "hola"


    idiomas = {

        'Inglés': {'codigo': 'en', 'nltk': 'english'},
        'Francés': {'codigo': 'fr', 'nltk': 'french'},
        'Aleman': {'codigo': 'de', 'nltk': 'german'},
        'italiano': {'codigo': 'it', 'nltk': 'italian'},
        'portugues': {'codigo': 'pt', 'nltk': 'portugues'},
    }

    traductor = GoogleTranslator(source='auto', target='en')
    traducto = GoogleTranslator(source='auto', target='fr')
    traduct = GoogleTranslator(source='auto', target='de')
    traduc = GoogleTranslator(source='auto', target='it')
    tradu = GoogleTranslator(source='auto', target='pt')

    if seleccion == "i":
        print(traductor.translate(texto))
    if seleccion == "f":
        print(traducto.translate(texto))
    if seleccion == "y":
        print(traduc.translate(texto))
    if seleccion == "a":
        print(traduct.translate(texto))
    if seleccion == "p":
        print(tradu.translate(texto))
    if seleccion == "x":
        break
    else:
        print("No se le entendio a lo que dijiste intentelo de nuevo")

import

Se usa para cargar módulos o bibliotecas que contienen funciones, clases o variables que quieres usar en tu código.

GoogleTranslator
